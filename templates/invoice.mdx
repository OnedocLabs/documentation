```javascript components.tsx
export const LineSeparation = () => {

export const PartySection = ({
  party,
  buyer,
}: {
  party: Party;
  buyer: boolean;
}) => {
  let partyDetails = Object.entries(party).map(([key,value]) => (
    <p className="p-0 mb-1">{value}</p>
  ));

  return buyer ? (
    <div>
      <p className="p-0 mb-1">
        <b>Bill to:</b>
      </p>
      {partyDetails}
    </div>
  ) : (
    <Right>{partyDetails}</Right>
  );
};

export const ItemTable = ({ items }: { items: Item[] }) => {
  const content = items.map((a) => {
    return (
      <React.Fragment key={items.indexOf(a)}>

          <td className="py-2">{items.indexOf(a)}</td>

          <td className="py-2">{a.quantity}</td>
```

## How to use our template ?

Render the template by running `npx tsx render.tsx`, your document will be written in your folder.

```javascript render.tsx
import React from "react";
import { Invoice } from "./invoice.tsx";
import { Onedoc } from "@onedoc/client";
import { renderToString } from "react-dom/server";
import postcss from "postcss";
import tailwindcss from "tailwindcss";
import postcssColorFunctionalNotation from "postcss-color-functional-notation";
import fs from "fs";


const ONEDOC_API_KEY = "XXXX-XXXX-XXXX-XXXX-XXXX"; // replace with your api key
const onedoc = new Onedoc(ONEDOC_API_KEY);

(async () => {

  const data: ComponentProps<typeof Invoice> = {
    id: 1029,
    seller: {
      name: "Onedoc, Inc,",
      address:"1600 Pennsylvania Avenue NW,",
      city:"Washington",
      ZIPCode:"DC 20500",
      country:"United States of America"
    },
    buyer: {
      name: "Titouan Launay",
      address:"172 Foxcol",
      city:"Gotham City,",
      ZIPCode:"NJ 12345",
      country:"United States of America"
    },
    date: new Date(),
    dateString: new Intl.DateTimeFormat('en-US', { 
      year: 'numeric', 
      month: 'long', 
      day: 'numeric' 
    }).format(new Date()),
    items: [
      {
        name: "Onedoc Startup Plan",
        price: 100,
        quantity: 1,
        total: 100,
      },
      {
        name: "Onedoc Document Hosting",
        price: 0,
        quantity: 3462,
        total: 0,
      },
    ],
  };

  const html = renderToString(<Invoice {...data} />);
  const css = await postcss([
    tailwindcss({
      content: [{ raw: html, extension: "html" }],
    }),
    postcssColorFunctionalNotation,
  ]).process(`@tailwind base;@tailwind components;@tailwind utilities;`, {
    from: undefined,
  });

  // load the file in @onedoc/react-print/dist/style.css to a text string
  // first start by resolving the path to @onedoc/react-print by using the node resolution
  const reactPrintPath = require.resolve("@onedoc/react-print/dist/index.css");

  const reactPrintStyles = fs.readFileSync(reactPrintPath);

  const { file } = await onedoc.render({
    html,
    test:false,
    save:false,
    assets: [
      {
        path: "./style.css",
        content: css.toString(),
      },
      {
        path: "./react-print.css",
        content: reactPrintStyles.toString(),
      },
    ],
  });
  
  fs.writeFileSync("./invoice.pdf", Buffer.from(file));

})();
```
