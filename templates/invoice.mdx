```javascript index_react.js
import { Frontmatter, formatters } from "../../components/frontmatter";
import { Logo } from "../../components/logo";
import { PageBottom } from "../../components/shell";

export const components = {
  p: (props) => <p className="p-0" {...props} />,
  hr: (props) => <div className="h-px bg-gray-300 my-4" {...props} />,
  tr: (props) => <tr className="border-b border-gray-300" {...props} />,
  table: (props) => <table className="w-full my-12" {...props} />,
  th: (props) => <th className="text-left font-bold py-2" {...props} />,
  td: (props) => <td className="py-2" {...props} />,
};

export const Right = ({ children }) => {
  return <div className="text-right">{children}</div>;
};

export const Template = ({ children }) => {
  return (
    <div>
      <div className="flex justify-between items-end pb-4">
        <div>
          <h1 className="text-2xl font-bold">
            Invoice #<Frontmatter field="invoice" />
          </h1>
          <p className="text-xs">
            <Frontmatter
              field="date"
              formatter={formatters.date("en-US", {
                year: "numeric",
                month: "long",
                day: "numeric",
              })}
            />
          </p>
        </div>
        <Logo className="h-6" />
      </div>
      {children}
      <PageBottom>
        <components.hr />
        <div className="text-gray-400 text-sm">
          Invoice #<Frontmatter field="invoice" />
        </div>
      </PageBottom>
    </div>
  );
};
```
