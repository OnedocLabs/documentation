## How to use our template ?

```javascript Render a template
import InvoiceTemplate from "./invoiceTemplate.js"
import { renderToString } from 'react-dom/server';
import { Onedoc } from "@onedoc/client"

let onedoc = new Onedoc(process.env.ONEDOC_API_KEY)

const data = {"invoiceNumber":"23213146", ...} // load data for template

let doc = {
        html: renderToString(< InvoiceTemplate props={data} />),
        metadata: {
          title: "Document title",
          author: "Batman",
          subject: "A doc with a title",
        },
        assets:[
                { path:"./App.tailwindcompiled.module.css", content: fs.readFileSync(path.join(process.cwd(), 'public', 'App.tailwindcompiled.module.css')).toString() } // example for next js with a stylesheet in /public
              ]
       };

const pdf = (await onedoc.render(doc, true)).file;

const pdfBuffer = Buffer.from(pdf); // The PDF can be written to your file system or returned in an endpoint as is.

```
