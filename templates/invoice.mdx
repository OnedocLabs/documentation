export const components = {
  p: (props) => <p className="p-0" {...props} />,
  hr: (props) => <div className="h-px bg-gray-300 my-4" {...props} />,
  tr: (props) => <tr className="border-b border-gray-300" {...props} />,

export const Right = ({ children }) => {
  return <div className="text-right">{children}</div>;
};

export const InvoiceHeader = ({ props }) => {
  return (
    <div>
      <h1 className="text-2xl font-bold">Invoice #{props.invoiceNumber}</h1>
      <p className="text-xs">{props.date}</p>
    </div>
  );
};

export const BillingCompany = ({ props }) => {
  return (
    <Right>
      <components.p>
        <b>{props.companyName}</b>
      </components.p>
      <components.p>{props.companyAddress},</components.p>
      <components.p>{props.city},</components.p>
      <components.p>{props.ZIPCode},</components.p>
      <components.p>{props.country}</components.p>
    </Right>
  );
};

export const CustomerInfo = ({ props }) => {
  return (
    <div>
      <components.p>
        <b>Bill to:</b>{" "}
      </components.p>
      <components.p>{props.customerName}</components.p>
      <components.p>{props.customerAddress},</components.p>
      <components.p>
        {props.city} {props.ZIPCode},
      </components.p>
      <components.p>{props.country}</components.p>
    </div>
  );
};

export const ProductTable = ({ props }) => {
  const articles = props;
  const content = articles.map((a) => {
    return (
      <React.Fragment key={articles.indexOf(a)}>
        <components.tr>
          <components.td>{articles.indexOf(a)}</components.td>
          <components.td>{a.description}</components.td>
          <components.td>${a.unitPrice}</components.td>
          <components.td>{a.quantity}</components.td>
          <components.td>${a.totalPrice}</components.td>
        </components.tr>
      </React.Fragment>
    );
  });
  const totalPrice = articles
    .map((a) => {
      return a.totalPrice;
    })
    .reduce((accumulator, currentValue) => {
      return accumulator + currentValue;
    }, 0);
  return (
    <components.table>
      <components.tr>
        <components.th>Item</components.th>
        <components.th>Description</components.th>
        <components.th>Unit Price</components.th>
        <components.th>Quantity</components.th>
        <components.th>Amount</components.th>
      </components.tr>

      {content}

      <components.tr>
        <components.th></components.th>
        <components.th>Total</components.th>
        <components.th></components.th>
        <components.th></components.th>
        <components.th>${totalPrice}</components.th>
      </components.tr>
    </components.table>
  );
};

export const InvoiceTemplate = ({ props }) => {
  const monthNames = [
    "January",
    "February",
    "March",
    "April",
    "May",
    "June",
    "July",
    "August",
    "September",
    "October",
    "November",
    "December",
  ];

  return (
    <div>
      <div className="flex justify-between items-end pb-4">
        <InvoiceHeader props={props} />
        <Logo className="h-6" />
      </div>
      <BillingCompany props={props.BillingCompany} />
      <components.hr></components.hr>
      <CustomerInfo props={props.CustomerInfo} />
      <components.hr></components.hr>
      <components.p>
        All items below correspond to work completed in the month of{" "}
        {monthNames[new Date().getMonth()]} {new Date().getFullYear()}. Payment
        is due within 15 days of receipt of this invoice.
        <Footnote>This includes non-business days.</Footnote>
      </components.p>
      <ProductTable props={props.articles} />
      <div className="bg-blue-100 p-3 rounded-md border-blue-300 text-blue-800 text-sm">
        On {props.date}, Onedoc users will be upgraded free of charge to our new
        cloud offering.
      </div>
      <PageBottom>
        <components.hr />
        <div className="text-gray-400 text-sm">
          Invoice #{props.invoiceNumber}
        </div>
      </PageBottom>
    </div>
  );
};

import InvoiceTemplate from "./invoice";

import { Onedoc } from "@onedoc/client";
import { readFileSync, writeFileSync } from "fs";

let onedoc = new Onedoc(process.env.ONEDOC_API_KEY);

const data = {}; // load data for template

let doc = {
  html: renderToString(<InvoiceTemplate props={data} />),
  metadata: {
    title: "Document title",
    test: true,

  assets: [
    {
      path: "./tailwindCompiled.css",

const { file, link, error, info } = await onedoc.render(doc);
